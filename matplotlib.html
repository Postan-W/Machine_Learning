%matplotlib inline
#matplotlib中解决中文显示问题\
mat.rcParams['font.sans-serif'] = ['SimHei']
mat.rcParams['axes.unicode_minus'] = False
x = nup.arange(-1.57,1.57,0.2)
#创建画板
plt.figure()
plt.plot(x,nup.sin(x),color ='brown',linestyle = 'dashed',marker = 'D',label = '正弦函数')
plt.plot(x,nup.cos(x),color ='black',linestyle = 'dashed',marker = 'D',label = '正弦函数')
#生成默认图例(左上角)
plt.legend()
#给坐标轴设置标签
plt.xlabel('x轴')
plt.ylabel('y轴')
#给图形设置标题
plt.title('正弦函数与余弦函数')
#设置坐标轴的数值范围
plt.xlim(0,3)
plt.ylim(-0.8,0.8)
#设置坐标轴的刻度(第二个参数是显示的形式，没有计算方面的意义)
plt.xticks([1,2,3,4],[str(i)+'月' for i in range(1,4)])
plt.yticks([-1,0,1,2])
#获取坐标轴信息
aexs = plt.gca()
#设置边框
aexs.spines['top'].set_color('black')
aexs.spines['right'].set_color('black')
plt.show()
